# <type>(<scope>): <subject>
#
# types: feat, fix, docs, style, refactor, perf, test, build, ci, chore, revert
# scope: optional; package or area, e.g., apps/web, packages/api
# subject: imperative, lower-case, no trailing period, <= 72 chars
